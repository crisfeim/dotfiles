git:
    gitbig:
        Muestra los objetos más grandes del repositorio git con su tamaño, hash y nombre de archivo — útil para encontrar qué está engordando el repo:

        git verify-pack -v .git/objects/pack/*.idx | sort -k 3 -n | tail -20 | while read hash type size rest; do
        name=$(git rev-list --objects --all | grep "^$hash" | awk '{print $2}')
        echo "$size\t$hash\t${name:-unknown}"
        done
fossil:
	create repo:
		git fast-export --all --reencode=yes > export.txt
		fossil clone https://usuario@chiselapp.com/user/usuario/repository/repo local.db
		fossil import --git --incremental local.db < export.txt
		fossil push -R local.db
	eliminar un archivo:
	    fossil export --git > export.txt
		git init repo-git
        cd repo-git
        git fast-import < ../export.txt
        git filter-branch --tree-filter 'rm -rf node_modules' HEAD
        git for-each-ref --format="delete %(refname)" refs/original | git update-ref --stdin
        git reflog expire --expire=now --all
        git gc --prune=now --aggressive
        git fast-export --all > ../export-clean.txt
        cd ..
        fossil import --git repo-clean.fossil < export-clean.txt
